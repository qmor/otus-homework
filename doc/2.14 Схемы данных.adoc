=== Схемы данных

В этом разделе представлены схемы баз данных для основных сущностей системы.

==== PostgreSQL Schema

[source,sql]
----
-- Пользователи
CREATE TABLE users (
    user_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    full_name VARCHAR(255) NOT NULL,
    avatar_url VARCHAR(512),
    status VARCHAR(50) DEFAULT 'active', -- active, blocked
    role VARCHAR(50) DEFAULT 'user', -- admin, moderator, user
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Групповые чаты и каналы
CREATE TABLE group_chats (
    chat_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    type VARCHAR(50) NOT NULL, -- public, private
    owner_id UUID REFERENCES users(user_id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    is_closed BOOLEAN DEFAULT FALSE,
    max_participants INTEGER DEFAULT 500,
    description TEXT,
    avatar_url VARCHAR(512)
);

-- Личные чаты
CREATE TABLE direct_chats (
    chat_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user1_id UUID REFERENCES users(user_id) NOT NULL,
    user2_id UUID REFERENCES users(user_id) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(user1_id, user2_id)
);

-- Сообщения
CREATE TABLE messages (
    message_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    chat_id UUID NOT NULL,
    chat_type VARCHAR(50) NOT NULL, -- direct, group
    sender_id UUID REFERENCES users(user_id) NOT NULL,
    text TEXT,
    media_id UUID,
    sent_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    is_edited BOOLEAN DEFAULT FALSE,
    is_deleted BOOLEAN DEFAULT FALSE
);

-- Медиа-файлы
CREATE TABLE media (
    media_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(user_id) NOT NULL,
    file_key VARCHAR(512) NOT NULL, -- S3 key
    file_size BIGINT NOT NULL,
    mime_type VARCHAR(100) NOT NULL,
    uploaded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Участники групповых чатов
CREATE TABLE chat_participants (
    user_chat_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(user_id) NOT NULL,
    chat_id UUID REFERENCES group_chats(chat_id) NOT NULL,
    role VARCHAR(50) DEFAULT 'member', -- admin, moderator, member
    last_read_message_id UUID REFERENCES messages(message_id),
    is_muted BOOLEAN DEFAULT FALSE,
    joined_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(user_id, chat_id)
);

-- Уведомления
CREATE TABLE notifications (
    notif_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(user_id) NOT NULL,
    message_id UUID REFERENCES messages(message_id),
    type VARCHAR(50) NOT NULL, -- new_message, mention, group_invite
    is_read BOOLEAN DEFAULT FALSE,
    sent_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    delivery_status VARCHAR(50) DEFAULT 'pending' -- sent, delivered, failed
);

-- Индексы для оптимизации запросов
CREATE INDEX idx_messages_chat_id ON messages(chat_id);
CREATE INDEX idx_messages_sender_id ON messages(sender_id);
CREATE INDEX idx_messages_sent_at ON messages(sent_at);
CREATE INDEX idx_chat_participants_user_id ON chat_participants(user_id);
CREATE INDEX idx_chat_participants_chat_id ON chat_participants(chat_id);
CREATE INDEX idx_notifications_user_id ON notifications(user_id);
CREATE INDEX idx_notifications_is_read ON notifications(is_read);
----
